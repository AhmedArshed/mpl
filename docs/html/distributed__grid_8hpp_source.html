63  throw invalid_dim();
64 #endif
65  auto C_size=C.dims();
66  auto C_coord=C.coords();
67  size_type vol=1;
68  for (std::size_t i=0; i<dim; ++i) {
69  gbegin_[i]=gbegin(gsize_[i], C_size[i], C_coord[i]);
70  gend_[i]=gend(gsize_[i], C_size[i], C_coord[i]);
71  size_[i]=gend_[i]-gbegin_[i];
72  oend_[i]=size_[i]+2*overlap_[i];
73  vol*=size_[i]+2*overlap_[i];
74  }
75  v.resize(vol);
76  for (std::size_t i=0; i<dim; ++i) {
77  typename subarray_layout<T>::parameter par_m_l, par_m_r, par_b_l, par_b_r;
78  for (std::size_t j=dim-1; true; --j) {
79  if (j==i) {
80  par_m_l.add(oend_[j], overlap_[j], 0);
81  par_m_r.add(oend_[j], overlap_[j], size_[j]+overlap_[j]);
82  par_b_l.add(oend_[j], overlap_[j], overlap_[j]);
83  par_b_r.add(oend_[j], overlap_[j], size_[j]);
84  } else {
85  par_m_l.add(oend_[j], size_[j], overlap_[j]);
86  par_m_r.add(oend_[j], size_[j], overlap_[j]);
87  par_b_l.add(oend_[j], size_[j], overlap_[j]);
88  par_b_r.add(oend_[j], size_[j], overlap_[j]);
89  }
90  if (j==0)

92  }

96  right_border_layout_.push_back(subarray_layout<T>(par_b_r));
