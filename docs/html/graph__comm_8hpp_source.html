51  explicit graph_communicator(const communicator &old_comm,
52  const edge_set &es,
53  bool reorder=true) {
54  int nodes=0;
55  for (const auto &e : es)
56  nodes=std::max({ nodes, e.first+1, e.second+1 });
57  int node=0;
58  int degree=0;
59  std::vector<int> edges, index(nodes, 0);
60  for (const auto &e : es) {
61  while (e.first>node) {
62  ++node;
63  degree=0;
64  }
65  edges.push_back(e.second);
66  index[e.first]+=1;
67  }
68  std::partial_sum(index.begin(), index.end(), index.begin());
69  MPI_Graph_create(old_comm.comm, nodes, index.data(), edges.data(), reorder, &comm);
70  }
