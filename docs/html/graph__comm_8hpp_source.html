51  const edge_set &es,
52  bool reorder=true) {
53  int nodes=0;
54  for (const auto &e : es)
55  nodes=std::max({ nodes, e.first+1, e.second+1 });
56  int node=0;
57  int degree=0;
58  std::vector<int> edges, index(nodes, 0);

60  while (e.first>node) {
61  ++node;
62  degree=0;

64  edges.push_back(e.second);

66  }
